1.2 Dataset

The dataset is constructed from Daniel's original CSV file to make everything fully replicable. I did some cleanup to make the code cleaner and I centered some variables for the interactions in the regressions. cpop is in 1000 habitants. cvc and cpvcu are percentages. crainmean and crainmax are daily rainfall based on monthly data. cage is in decades. 

raw <- read.csv("Data_For_R.csv")
center <- function(x, na.rm = T){
    return(x - mean(x, na.rm = na.rm))
}
names(raw) <- c("CC","CCab","Year","Blockages","BLOCKAGE",
  "COLLAPSED", "CUSTOMER RESONSIBILITY","DAMAGE","FAT","FATEXT",
  "FOREIGN","NO FAULT","OPERROR","OTHER","OVERFLOW",
  "RAGS","ROOTS","ROOTSEXT","ROOTSLIGHT","SAND",
  "SANDROOTS","UNKNOWN","PrevMain","SchedPrev",
  "ConMon","CBM","CorrMaint","BL_Length","BL_100km","BL_Age",
  "BL_Pop","Length","PVC_U","VC","RC","AC","GRP_CC",
  "PVC","S","GRP_FW","RCPL","InstallYr","YrsFromInstall",
  "Population","RainMax","RainMean")
ds <- raw %>% tbl_df() %>%
    filter(!is.na(Year)) %>%
    select(center = CC, year = Year, block = Blockages,
           prev = PrevMain, sched = SchedPrev, conmon = ConMon,
           cbm = CBM, corr = CorrMaint, length = Length,
           age = InstallYr, pvcu = PVC_U, vc = VC, rc = RC,
           pvc = PVC, ac = AC, grp = GRP_CC, pop = Population,
           rainmax = RainMax, rainmean = RainMean) %>%
    mutate(length = length/1000, age = year - age,
           block_km = block*100/length, cage = center(age)/10,
           clength = center(length), cpvcu = center(pvcu),
           cvc = center(pvc), crainmax = center(rainmax)/30,
           crainmean = center(rainmean)/30, cpop = center(pop)/1000)
